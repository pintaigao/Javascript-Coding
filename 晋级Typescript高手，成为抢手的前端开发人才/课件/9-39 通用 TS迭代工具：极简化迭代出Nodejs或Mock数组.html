需求：写一个通用 TS 工具类，要求实现如下：

  1. 可用于迭代模拟后台 mock 中的数组或 nodejs 框架从数据表取出的数组数据。

  2. 可以移植到任意一个项目中，极简化迭代代码.

  3. 模拟后台数据表对应的类，现在要求按照上述要求 用 TS 工具类迭代出 stuArr

 class Student {
  constructor(stuname: string) {
    this.stuname = stuname
  }
  stuname!: string
  study() {

    console.log("学生:", this.stuname)
  }
  eat() {
    console.log(this.stuname + "吃饭");
  }
}
let stuArr: Array<Student> = []
stuArr.push(new Student("wangwu"))
stuArr.push(new Student("lisi"))
